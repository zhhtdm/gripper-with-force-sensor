147    def set_force_callback(
148            self, 
149            callback: Optional[Callable[[float], None]] = None,
150            callback_threshold:float = 0.1, 
151            single_callback_mode: bool = False):
152        """
153        注册并设置力值的回调函数，也可以不注册回调函数，只用 force() 轮询

155        - `callback` : 注册力值变化回调函数`(Callable[[float], None])`, 接收一个`float`形参(力值)
156        - `callback_threshold` : 力绝对值或力的变化绝对值超过此值触发回调，单位 kg
157        - `single_callback_mode` : 此值为真则: `callback_threshold`为力的绝对值阈值，为假: `callback_threshold`为力的变化绝对值阈值。达到阈值触发回调
158        """

161        self._single_callback_mode = single_callback_mode

163    def _on_atc_message(self, msg: str):
164        try:
165            res = json.loads(msg)
166            if "force_sensor_zeroing" in res:
167                res.pop("force_sensor_zeroing")
168            self._status.update(res)
169
170            if "force" in res and self._force_callback:
171                force_val:float = res["force"]
172                if self._single_callback_mode:
173                    if abs(force_val) >= self._callback_threshold and self._last_force < self._callback_threshold:
174                        self._force_callback(force_val)
175                    self._last_force = force_val
176                else:
177                    if abs(self._last_force - force_val) >= self._callback_threshold:
178                        self._last_force = force_val
179                        self._force_callback(force_val)
180                
181
182        except Exception as e:
183            logging.error(f'[{self._name}] 抓手响应解析失败 : error: {e} | msg : {msg}')
